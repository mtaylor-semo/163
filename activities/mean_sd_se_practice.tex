	\bigskip
	
	\bigskip
	
	Mean:
	\begin{equation*}
	\overline{Y} = \dfrac{\sum\limits^n_{i=1} Y_i}{n}
	\end{equation*}
	
	\bigskip
	
	Standard Deviation:
	\begin{equation*}
	 s = \sqrt{\dfrac{\sum\left(Y_i - \overline{Y}\right)^2}{n-1}}
	\end{equation*}

	\bigskip

	Standard Error:
	\begin{equation*}
	S_{\overline{Y}} = \dfrac{s}{\sqrt{n}}
	\end{equation*}
