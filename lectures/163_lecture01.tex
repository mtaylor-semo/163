\begin{frame}[t]{Evolution summary}

	\hangpara \highlight{Population:} a group of individuals of the \emph{same species} together in the same area.
	
	\pause\hangpara \highlight{Time:} measurable differences can take dozens to hundreds or thousands of generations.
	
	\pause\hangpara \highlight{Genetic change:} allele frequencies change in the population over time.
	
	\pause\hangpara Evolution of the genotype causes evolution of the phenotype.

\end{frame}
%
\begin{frame}[t]{What causes allele frequencies to change in a population?}

	\hangpara \highlight{Mutation},
	
	\hangpara \highlight{migration},
	
	\hangpara \highlight{genetic drift},

	\hangpara \highlight{non-random mating}, and
	
	\hangpara \highlight{natural selection}.

	\hangpara If \emph{none} of these occur, then a population will not evolve. 
	
\end{frame}
