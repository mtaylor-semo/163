\begin{frame}[t]{Natural selection leads to descent with modification.}


	\hangpara \textcolor<2-4>{gray}{Natural selection causes variation in \only<1,4>{\highlight}{reproductive success} among individuals that vary in  \only<1>{\highlight}{heritable traits.}}

	\pause
	
	\hangpara	 \textcolor<3->{gray}{Traits that increase reproductive success are called  \only<2>{\highlight}{adaptations.}}
	
	\pause 
	
	\hangpara	 \textcolor<4->{gray}{Adaptations are traits that have been  \only<3>{\highlight}{modified} from the ancestral form.}

	\pause 
		
	\vspace{3ex}

	\hangpara Let us study variation in reproductive success more closely.


\end{frame}

\begin{frame}[t]{True or false?}

	\hangpara \textcolor<2->{gray}{Darwin coined the phrase “survival of the fittest.”}
	
	\pause
	
	\vspace*{2ex}
	
	\hangpara\alt<handout>{}{\highlight{False.} The phrase was coined by Herbert Spencer.}
	
	\hangpara\alt<handout>{}{Darwin did not use the phrase until the 5th edition of \textit{On the Origin of Species.}}

\end{frame}
